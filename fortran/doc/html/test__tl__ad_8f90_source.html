149 FUNCTION gasdev(idum)
150  INTEGER :: idum
151  REAL(KIND=8) :: gasdev,ran2
152  ! USES ran2
153  INTEGER :: iset
154  REAL(KIND=8) :: fac,gset,rsq,v1,v2
155  SAVE iset,gset
156  DATA iset/0/
157  if (idum.lt.0) iset=0
158  if (iset.eq.0) then
159 1 v1=2.d0*ran2(idum)-1.
160  v2=2.d0*ran2(idum)-1.
161  rsq=v1**2+v2**2
162  if (rsq.ge.1.d0.or.rsq.eq.0.d0) goto 1
163  fac=sqrt(-2.*log(rsq)/rsq)
164  gset=v1*fac
165  gasdev=v2*fac
166  iset=1
167  else
168  gasdev=gset
169  iset=0
170  endif
171  return
172 END FUNCTION gasdev
