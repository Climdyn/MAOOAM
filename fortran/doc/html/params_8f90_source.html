133  !> Parameters initialisation routine
134  SUBROUTINE init_params
135  INTEGER, DIMENSION(2) :: s
136  INTEGER :: i
137  CALL init_nml
138 
139  !---------------------------------------------------------!
140  ! !
141  ! Computation of the dimension of the atmospheric !
142  ! and oceanic components !
143  ! !
144  !---------------------------------------------------------!
145 
146  natm=0
147  DO i=1,nbatm
148  IF (ams(i,1)==1) THEN
149  natm=natm+3
150  ELSE
151  natm=natm+2
152  ENDIF
153  ENDDO
154  s=shape(oms)
155  noc=s(1)
156 
157  ndim=2*natm+2*noc
158 
159  !---------------------------------------------------------!
160  ! !
161  ! Some general parameters (Domain, beta, gamma, coupling) !
162  ! !
163  !---------------------------------------------------------!
164 
165  pi=dacos(-1.d0)
166  l=scale/pi
167  phi0=phi0_npi*pi
168  lr=sqrt(gp*h)/f0
169  g=-l**2/lr**2
170  betp=l/rra*cos(phi0)/sin(phi0)
171  rp=r/f0
172  dp=d/f0
173  kd=k*2
174  kdp=kp
175 
176  !-----------------------------------------------------!
177  ! !
178  ! DERIVED QUANTITIES !
179  ! !
180  !-----------------------------------------------------!
181 
182  cpo=co/(go*f0) * rr/(f0**2*l**2)
183  lpo=lambda/(go*f0)
184  cpa=ca/(ga*f0) * rr/(f0**2*l**2)/2 ! Cpa acts on psi1-psi3, not on theta
185  lpa=lambda/(ga*f0)
186  sbpo=4*sb*to0**3/(go*f0) ! long wave radiation lost by ocean to atmosphere space
187  sbpa=8*epsa*sb*ta0**3/(go*f0) ! long wave radiation from atmosphere absorbed by ocean
188  lsbpo=2*epsa*sb*to0**3/(ga*f0) ! long wave radiation from ocean absorbed by atmosphere
189  lsbpa=8*epsa*sb*ta0**3/(ga*f0) ! long wave radiation lost by atmosphere to space & ocea
190 
191 
192  END SUBROUTINE init_params
